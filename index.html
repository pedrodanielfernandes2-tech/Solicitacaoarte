// ============================================================ // ASSEMBLEIA DE DEUS LOUVEIRA — Apps Script // Grava dados na planilha, salva fotos no Drive e envia email // // CONFIGURAÇÃO: // 1. Cole este código no Apps Script (Extensões > Apps Script) // 2. Altere o EMAIL_DESTINO abaixo // 3. Salve (Ctrl+S) // 4. Implantar > Gerenciar implantações > lápis > Nova versão > Implantar // Executar como: Eu | Quem acessa: Qualquer pessoa // 5. Na primeira execução, autorize as permissões solicitadas // ============================================================ var EMAIL_DESTINO = 'secretaria@adlouveira.com.br'; var PASTA_FOTOS = 'AD Louveira — Fotos Solicitações'; // pasta criada automaticamente no Drive // ── Ponto de entrada POST ────────────────────────────────── function doPost(e) { try { var p = lerParametros(e); var sheet = getSheet(); var row = gravarPlanilha(sheet, p); var linksForotos = salvarFotos(p, row); enviarEmail(p, row, linksForotos); return ContentService .createTextOutput(JSON.stringify({ status: 'ok', linha: row })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { GmailApp.sendEmail( EMAIL_DESTINO, '[ERRO] Solicitação de Arte — AD Louveira', 'Erro ao processar solicitação:\n\n' + err.toString() ); return ContentService .createTextOutput(JSON.stringify({ status: 'error', msg: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // ── Lê parâmetros do POST ────────────────────────────────── function lerParametros(e) { var p = {}; if (e && e.parameter && Object.keys(e.parameter).length > 0) { p = e.parameter; } else if (e && e.postData && e.postData.contents) { e.postData.contents.split('&').forEach(function(pair) { var kv = pair.split('='); if (kv.length >= 2) { var key = decodeURIComponent(kv[0].replace(/\+/g, ' ')); var val = decodeURIComponent(kv.slice(1).join('=').replace(/\+/g, ' ')); p[key] = val; } }); } return p; } // ── Acessa ou cria a planilha ────────────────────────────── function getSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getActiveSheet(); if (sheet.getLastRow() === 0) { var headers = [ 'Nº','Data Solicitação','Congregação','Solicitante', 'Nome do Trabalho','Local','Tema','Referência Bíblica', 'Datas/Horários','Pregadores/Ministros', 'Paleta de Cores','Mais Informações','Observações', 'Links das Fotos','Timestamp' ]; sheet.appendRow(headers); var hRange = sheet.getRange(1, 1, 1, headers.length); hRange.setBackground('#3aafa9') .setFontColor('#ffffff') .setFontWeight('bold') .setHorizontalAlignment('center'); sheet.setFrozenRows(1); sheet.setColumnWidth(1, 40); } return sheet; } // ── Grava linha na planilha ──────────────────────────────── function gravarPlanilha(sheet, p) { var rowNum = sheet.getLastRow(); // será a linha do novo registro (sem cabeçalho = nº sequencial) sheet.appendRow([ rowNum, // Nº sequencial p.data_solicitacao || '', p.congregacao || '', p.solicitante || '', p.nome_trabalho || '', p.local || '', p.tema || '', p.referencia || '', p.datas || '', p.pregadores || '', p.paleta_cores || '', p.mais_info || '', p.observacoes || '', '', // Links das fotos — preenchido depois new Date().toLocaleString('pt-BR') ]); sheet.autoResizeColumns(1, 15); return rowNum; // linha onde foi gravado (para atualizar depois) } // ── Salva fotos no Google Drive ──────────────────────────── function salvarFotos(p, rowNum) { var links = []; // Cria pasta raiz e subpasta desta solicitação (só se houver arquivos) var temFotos = (p.fotos_json && p.fotos_json.trim() !== '') || (p.paleta_fotos_json && p.paleta_fotos_json.trim() !== ''); if (!temFotos) return []; var pastaRaiz = getOrCreateFolder(PASTA_FOTOS); var subNome = 'Solicitação #' + rowNum + ' — ' + (p.nome_trabalho || 'Sem nome') + ' (' + (p.data_solicitacao || '') + ')'; var subPasta = pastaRaiz.createFolder(subNome); // ── Fotos dos pregadores ── if (p.fotos_json && p.fotos_json.trim() !== '') { var fotos; try { fotos = JSON.parse(p.fotos_json); } catch(e) { fotos = []; } var subPreg = subPasta.createFolder('Fotos Pregadores'); fotos.forEach(function(foto) { try { var b64 = foto.data.replace(/^data:image\/[a-z]+;base64,/, ''); var bytes = Utilities.base64Decode(b64); var blob = Utilities.newBlob(bytes, 'image/jpeg', foto.nome || ('pregador_' + foto.pregador + '.jpg')); var file = subPreg.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); links.push('[Pregador ' + foto.pregador + '] ' + file.getUrl()); } catch(err) { links.push('[Pregador ' + foto.pregador + '] Erro: ' + err.message); } }); } // ── Imagens de referência de paleta ── if (p.paleta_fotos_json && p.paleta_fotos_json.trim() !== '') { var paleta; try { paleta = JSON.parse(p.paleta_fotos_json); } catch(e) { paleta = []; } var subPaleta = subPasta.createFolder('Referência Visual'); paleta.forEach(function(img, idx) { try { var b64 = img.data.replace(/^data:image\/[a-z]+;base64,/, ''); var bytes = Utilities.base64Decode(b64); var blob = Utilities.newBlob(bytes, 'image/jpeg', img.nome || ('referencia_' + (idx+1) + '.jpg')); var file = subPaleta.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); links.push('[Ref. Visual ' + (idx+1) + '] ' + file.getUrl()); } catch(err) { links.push('[Ref. Visual ' + (idx+1) + '] Erro: ' + err.message); } }); } // Atualiza coluna "Links das Fotos" na planilha if (links.length > 0) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); sheet.getRange(rowNum + 1, 14).setValue(links.join('\n')); } return links; } // ── Função auxiliar: pega ou cria pasta no Drive ─────────── function getOrCreateFolder(nome) { var folders = DriveApp.getFoldersByName(nome); if (folders.hasNext()) return folders.next(); return DriveApp.createFolder(nome); } // ── Envia email automático com resumo ────────────────────── function enviarEmail(p, rowNum, linksForotos) { var assunto = 'Nova Solicitação de Arte #' + rowNum + ' — ' + (p.nome_trabalho || 'Sem nome') + ' | ' + (p.congregacao || ''); var fotosTexto = linksForotos.length > 0 ? linksForotos.join('\n') : 'Nenhuma foto enviada'; var coresTexto = p.paleta_cores && p.paleta_cores.trim() !== '' ? p.paleta_cores : 'Não informada'; // ── Corpo em HTML ── var html = '
' + // cabeçalho '
' + '

Assembleia de Deus Louveira

' + '

Nova Solicitação de Arte #' + rowNum + '

' + '
' + // corpo '
' + bloco('📋 Identificação', [ ['Congregação', p.congregacao || '—'], ['Solicitante', p.solicitante || '—'], ['Data Solicitação', p.data_solicitacao || '—'] ]) + bloco('📅 Dados do Evento', [ ['Nome do Trabalho', p.nome_trabalho || '—'], ['Datas/Horários', (p.datas || '—').replace(/ \|\| /g, '
')], ['Local', p.local || '—'], ['Tema', p.tema || '—'], ['Ref. Bíblica', p.referencia || '—'] ]) + bloco('🎤 Pregadores / Ministros', [ ['Pregadores', (p.pregadores || '—').replace(/ \|\| /g, '
')] ]) + bloco('🎨 Referência Visual / Paleta', [ ['Arquivos', coresTexto] ]) + bloco('📎 Fotos (Google Drive)', [ ['Links', fotosTexto.replace(/\n/g, '
')] ]) + bloco('💬 Informações Complementares', [ ['Mais Informações', p.mais_info || '—'], ['Observações', p.observacoes || '—'] ]) + '
' + // rodapé '
' + 'Enviado automaticamente pelo formulário de solicitações — AD Louveira
' + 'Prazos: 60 dias (ideal) | 30 dias (mínimo)' + '
' + '
'; GmailApp.sendEmail(EMAIL_DESTINO, assunto, '', { htmlBody: html }); } // ── Helper: bloco de seção para o email ─────────────────── function bloco(titulo, campos) { var linhas = campos.map(function(c) { return '' + '' + c[0] + '' + '' + c[1] + '' + ''; }).join(''); return '
' + '
' + titulo + '
' + '' + linhas + '
' + '
'; } // ── Teste via GET ────────────────────────────────────────── function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ status: 'online', msg: 'Script ativo e pronto para receber dados!' })) .setMimeType(ContentService.MimeType.JSON); }
